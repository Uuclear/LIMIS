## 3. Model 定义规范

### 3.1 基础模型类

所有模型继承自 `BaseModel`:

```python
# /opt/limis/backend/core/models.py

from django.db import models
from django.conf import settings

class BaseModel(models.Model):
    """基础模型类，包含通用字段"""
    
    created_at = models.DateTimeField(
        auto_now_add=True,
        verbose_name='创建时间'
    )
    updated_at = models.DateTimeField(
        auto_now=True,
        verbose_name='更新时间'
    )
    created_by = models.ForeignKey(
        settings.AUTH_USER_MODEL,
        on_delete=models.SET_NULL,
        null=True,
        blank=True,
        related_name='created_%(class)s_set',
        verbose_name='创建人'
    )
    is_deleted = models.BooleanField(
        default=False,
        verbose_name='是否删除'
    )
    
    class Meta:
        abstract = True
        ordering = ['-created_at']
    
    def soft_delete(self):
        """软删除"""
        self.is_deleted = True
        self.save(update_fields=['is_deleted', 'updated_at'])
    
    def restore(self):
        """恢复删除"""
        self.is_deleted = False
        self.save(update_fields=['is_deleted', 'updated_at'])
```

```python
# apps/testing/models.py

from django.db import models
from django.conf import settings
from core.models import BaseModel


class TestCategory(BaseModel):
    """检测类别"""
    
    name = models.CharField(
        max_length=100,
        verbose_name='类别名称'
    )
    code = models.CharField(
        max_length=50,
        unique=True,
        verbose_name='类别编码'
    )
    parent = models.ForeignKey(
        'self',
        on_delete=models.CASCADE,
        null=True,
        blank=True,
        related_name='children',
        verbose_name='父级类别'
    )
    sort_order = models.IntegerField(
        default=0,
        verbose_name='排序'
    )
    
    class Meta:
        verbose_name = '检测类别'
        verbose_name_plural = verbose_name
        ordering = ['sort_order', 'code']
    
    def __str__(self):
        return self.name
    
    @property
    def full_name(self):
        """获取完整层级名称"""
        if self.parent:
            return f'{self.parent.full_name} / {self.name}'
        return self.name


class TestMethod(BaseModel):
    """检测方法"""
    
    name = models.CharField(
        max_length=200,
        verbose_name='方法名称'
    )
    standard_no = models.CharField(
        max_length=50,
        verbose_name='标准编号'
    )
    standard_name = models.CharField(
        max_length=200,
        verbose_name='标准名称'
    )
    category = models.ForeignKey(
        TestCategory,
        on_delete=models.PROTECT,
        related_name='methods',
        verbose_name='检测类别'
    )
    description = models.TextField(
        blank=True,
        verbose_name='方法描述'
    )
    is_active = models.BooleanField(
        default=True,
        verbose_name='是否启用'
    )
    
    class Meta:
        verbose_name = '检测方法'
        verbose_name_plural = verbose_name
        ordering = ['standard_no']
        indexes = [
            models.Index(fields=['standard_no']),
            models.Index(fields=['category', 'is_active']),
        ]
    
    def __str__(self):
        return f'{self.standard_no} - {self.name}'


class TestParameter(BaseModel):
    """检测参数"""
    
    method = models.ForeignKey(
        TestMethod,
        on_delete=models.CASCADE,
        related_name='parameters',
        verbose_name='检测方法'
    )
    name = models.CharField(
        max_length=100,
        verbose_name='参数名称'
    )
    code = models.CharField(
        max_length=50,
        verbose_name='参数编码'
    )
    unit = models.CharField(
        max_length=20,
        blank=True,
        verbose_name='单位'
    )
    precision = models.IntegerField(
        default=2,
        verbose_name='精度(小数位数)'
    )
    min_value = models.DecimalField(
        max_digits=12,
        decimal_places=4,
        null=True,
        blank=True,
        verbose_name='最小值'
    )
    max_value = models.DecimalField(
        max_digits=12,
        decimal_places=4,
        null=True,
        blank=True,
        verbose_name='最大值'
    )
    is_required = models.BooleanField(
        default=True,
        verbose_name='是否必检'
    )
    
    class Meta:
        verbose_name = '检测参数'
        verbose_name_plural = verbose_name
        ordering = ['method', 'code']
        unique_together = [['method', 'code']]
    
    def __str__(self):
        return f'{self.method.name} - {self.name}'
```

### 3.3 模型定义规范

1. **字段命名**: 使用小写字母和下划线，如 `sample_no`
2. **verbose_name**: 所有字段必须设置中文 verbose_name
3. **关联字段**: 使用 `on_delete=models.PROTECT` 保护关键数据
4. **索引**: 为常用查询字段添加索引
5. **字符串表示**: 实现 `__str__` 方法
6. **Meta 类**: 设置 verbose_name、ordering、indexes

### 3.4 模型方法示例

```python
class TestTask(BaseModel):
    """检测任务"""
    
    STATUS_CHOICES = [
        ('pending', '待分配'),
        ('assigned', '已分配'),
        ('testing', '检测中'),
        ('completed', '已完成'),
        ('reviewed', '已复核'),
    ]
    
    task_no = models.CharField(max_length=50, unique=True)
    sample = models.ForeignKey('samples.Sample', on_delete=models.PROTECT)
    test_method = models.ForeignKey(TestMethod, on_delete=models.PROTECT)
    status = models.CharField(max_length=20, choices=STATUS_CHOICES, default='pending')
    planned_date = models.DateField(null=True, blank=True)
    actual_date = models.DateField(null=True, blank=True)
    
    @property
    def is_overdue(self):
        """是否超期"""
        if self.planned_date and self.status not in ['completed', 'reviewed']:
            from datetime import date
            return date.today() > self.planned_date
        return False
    
    def can_assign(self):
        """是否可分配"""
        return self.status == 'pending'
    
    def assign(self, tester, equipment=None, planned_date=None):
        """分配任务"""
        if not self.can_assign():
            raise ValueError('当前状态不允许分配')
        self.assigned_tester = tester
        self.assigned_equipment = equipment
        self.planned_date = planned_date
        self.status = 'assigned'
        self.save()
    
    def start(self):
        """开始检测"""
        if self.status != 'assigned':
            raise ValueError('当前状态不允许开始检测')
        self.status = 'testing'
        self.actual_date = date.today()
        self.save()
    
    def complete(self):
        """完成检测"""
        if self.status != 'testing':
            raise ValueError('当前状态不允许完成')
        self.status = 'completed'
        self.save()
```

### 4.2 序列化器示例

```python
# apps/testing/serializers.py

from rest_framework import serializers
from core.serializers import BaseModelSerializer
from .models import TestCategory, TestMethod, TestParameter, TestTask


class TestCategorySerializer(BaseModelSerializer):
    """检测类别序列化器"""
    
    # 嵌套显示子类别
    children = serializers.SerializerMethodField()
    # 关联字段只读显示
    parent_name = serializers.CharField(source='parent.name', read_only=True)
    
    class Meta:
        model = TestCategory
        fields = [
            'id', 'name', 'code', 'parent', 'parent_name',
            'sort_order', 'children', 'created_at', 'updated_at',
        ]
        read_only_fields = ['id', 'created_at', 'updated_at']
    
    def get_children(self, obj):
        """获取子类别"""
        children = obj.children.filter(is_deleted=False)
        return TestCategorySerializer(children, many=True).data
    
    def validate_code(self, value):
        """验证编码唯一性"""
        if TestCategory.objects.filter(code=value).exclude(pk=self.instance.pk if self.instance else None).exists():
            raise serializers.ValidationError('类别编码已存在')
        return value


class TestMethodSerializer(BaseModelSerializer):
    """检测方法序列化器"""
    
    category_name = serializers.CharField(source='category.name', read_only=True)
    status_display = serializers.CharField(source='get_status_display', read_only=True)
    
    class Meta:
        model = TestMethod
        fields = [
            'id', 'name', 'standard_no', 'standard_name',
            'category', 'category_name', 'description', 'is_active',
            'created_at', 'updated_at',
        ]
        read_only_fields = ['id', 'created_at', 'updated_at']


class TestMethodDetailSerializer(TestMethodSerializer):
    """检测方法详情序列化器 - 包含参数"""
    
    parameters = TestParameterSerializer(many=True, read_only=True)
    
    class Meta(TestMethodSerializer.Meta):
        fields = TestMethodSerializer.Meta.fields + ['parameters']


class TestParameterSerializer(BaseModelSerializer):
    """检测参数序列化器"""
    
    class Meta:
        model = TestParameter
        fields = [
            'id', 'method', 'name', 'code', 'unit', 'precision',
            'min_value', 'max_value', 'is_required',
            'created_at', 'updated_at',
        ]
        read_only_fields = ['id', 'created_at', 'updated_at']


class TestTaskListSerializer(BaseModelSerializer):
    """检测任务列表序列化器"""
    
    # 关联对象字段
    sample_name = serializers.CharField(source='sample.name', read_only=True)
    sample_no = serializers.CharField(source='sample.sample_no', read_only=True)
    method_name = serializers.CharField(source='test_method.name', read_only=True)
    tester_name = serializers.SerializerMethodField()
    status_display = serializers.CharField(source='get_status_display', read_only=True)
    is_overdue = serializers.BooleanField(read_only=True)
    
    class Meta:
        model = TestTask
        fields = [
            'id', 'task_no', 'sample', 'sample_name', 'sample_no',
            'test_method', 'method_name',
            'assigned_tester', 'tester_name',
            'planned_date', 'actual_date', 'status', 'status_display',
            'is_overdue', 'created_at',
        ]
        read_only_fields = ['id', 'task_no', 'created_at']
    
    def get_tester_name(self, obj):
        """获取检测员姓名"""
        if obj.assigned_tester:
            return obj.assigned_tester.get_full_name() or str(obj.assigned_tester)
        return ''


class TestTaskDetailSerializer(TestTaskListSerializer):
    """检测任务详情序列化器"""
    
    test_method_detail = TestMethodSerializer(source='test_method', read_only=True)
    
    class Meta(TestTaskListSerializer.Meta):
        fields = TestTaskListSerializer.Meta.fields + [
            'test_parameter', 'assigned_equipment',
            'test_method_detail', 'remark', 'updated_at', 'created_by',
        ]


class TestTaskCreateSerializer(BaseModelSerializer):
    """检测任务创建序列化器"""
    
    class Meta:
        model = TestTask
        fields = [
            'sample', 'test_method', 'test_parameter',
            'assigned_tester', 'assigned_equipment',
            'planned_date', 'remark',
        ]
    
    def validate(self, attrs):
        """自定义验证"""
        # 验证检测参数属于所选方法
        method = attrs.get('test_method')
        parameter = attrs.get('test_parameter')
        if parameter and parameter.method_id != method.id:
            raise serializers.ValidationError({
                'test_parameter': '检测参数必须属于所选检测方法'
            })
        return attrs
```

### 4.3 序列化器规范

1. **分层设计**: 列表用精简版，详情用完整版
2. **关联字段**: 使用 `source` 获取关联对象属性
3. **只读字段**: 明确设置 `read_only_fields`
4. **验证方法**: 实现 `validate_{field}` 和 `validate` 方法
5. **嵌套序列化**: 合理使用嵌套，避免过度嵌套

## 5. ViewSet 编写

### 5.1 基础视图类

```python
# /opt/limis/backend/core/views.py

from rest_framework import viewsets, status
from rest_framework.response import Response
from rest_framework.permissions import IsAuthenticated
from django_filters.rest_framework import DjangoFilterBackend

class BaseModelViewSet(viewsets.ModelViewSet):
    """基础 ModelViewSet"""
    
    permission_classes = [IsAuthenticated]
    filter_backends = [DjangoFilterBackend]
    lookup_field = 'pk'
    
    # LIMS 模块标识，用于权限控制
    lims_module = None
    lims_action_map = {}
    
    def get_queryset(self):
        """默认过滤已删除数据"""
        qs = super().get_queryset()
        if hasattr(qs.model, 'is_deleted'):
            qs = qs.filter(is_deleted=False)
        return qs
    
    def perform_create(self, serializer):
        """创建时自动设置创建人"""
        if self.request.user.is_authenticated:
            serializer.save(created_by=self.request.user)
        else:
            serializer.save()
    
    def perform_destroy(self, instance):
        """软删除"""
        if hasattr(instance, 'soft_delete'):
            instance.soft_delete()
        else:
            instance.delete()
    
    def get_serializer_context(self):
        """添加请求上下文"""
        context = super().get_serializer_context()
        context['request'] = self.request
        return context
```

```python
    TestTaskAssignSerializer,

from .services import generate_task_no, assign_task, start_task, complete_task


class TestCategoryViewSet(BaseModelViewSet):
    """检测类别 ViewSet"""
    
    queryset = TestCategory.objects.filter(parent__isnull=True)
    serializer_class = TestCategorySerializer
    lims_module = 'testing'
    search_fields = ['name', 'code']


class TestMethodViewSet(BaseModelViewSet):
    """检测方法 ViewSet"""
    
    queryset = TestMethod.objects.select_related('category')
    lims_module = 'testing'
    filterset_class = TestMethodFilter
    search_fields = ['name', 'standard_no', 'standard_name']
    
    def get_serializer_class(self):
        """根据动作选择序列化器"""
        if self.action == 'retrieve':
            return TestMethodDetailSerializer
        return TestMethodSerializer
    
    def get_queryset(self):
        """根据资质过滤"""
        qs = super().get_queryset()
        if self.action in ('list', 'retrieve'):
            # 获取当前用户的资质范围
            from apps.quality.services import get_active_qualification_profile
            profile = get_active_qualification_profile()
            if profile:
                allowed_ids = profile.allowed_test_methods.values_list('id', flat=True)
                qs = qs.filter(id__in=allowed_ids)
        return qs
    
    @action(detail=True, methods=['get'])
    def parameters(self, request, pk=None):
        """获取方法的检测参数"""
        method = self.get_object()
        params = method.parameters.filter(is_deleted=False)
        serializer = TestParameterSerializer(params, many=True)
        return Response({
            'code': 200,
            'data': serializer.data
        })

        return TestTaskListSerializer
    
    def perform_create(self, serializer):
        """创建任务时生成编号"""
        task_no = generate_task_no()
        super().perform_create(serializer.save(task_no=task_no))
    
    @action(detail=True, methods=['post'])
    def assign(self, request, pk=None):
        """分配任务"""
        serializer = TestTaskAssignSerializer(data=request.data)
        serializer.is_valid(raise_exception=True)
        
        task = assign_task(
            task_id=int(pk),
            tester_id=serializer.validated_data['tester'],
            equipment_id=serializer.validated_data.get('equipment'),
            planned_date=serializer.validated_data.get('planned_date'),
        )
        
        return Response({
            'code': 200,
            'message': '分配成功',
            'data': TestTaskDetailSerializer(task).data,
        })
    
    @action(detail=True, methods=['post'])
    def start(self, request, pk=None):
        """开始检测"""
        task = start_task(int(pk))
        return Response({
            'code': 200,
            'message': '已开始检测',
            'data': TestTaskDetailSerializer(task).data,
        })
    
    @action(detail=True, methods=['post'])
    def complete(self, request, pk=None):
        """完成检测"""
        task = complete_task(int(pk))
        return Response({
            'code': 200,
            'message': '检测完成',
            'data': TestTaskDetailSerializer(task).data,
        })
    
    @action(detail=False, methods=['get'])
    def today_list(self, request):
        """今日任务列表"""
        user = request.user if request.query_params.get('mine') else None
        tasks = get_today_tasks(user=user)
        page = self.paginate_queryset(tasks)
        if page is not None:
            serializer = TestTaskListSerializer(page, many=True)
            return self.get_paginated_response(serializer.data)
        serializer = TestTaskListSerializer(tasks, many=True)
        return Response({'code': 200, 'data': serializer.data})
    
    @action(detail=False, methods=['get'])
    def overdue_list(self, request):
        """超期任务列表"""
        tasks = get_overdue_tasks()
        page = self.paginate_queryset(tasks)
        if page is not None:
            serializer = TestTaskListSerializer(page, many=True)
            return self.get_paginated_response(serializer.data)
        serializer = TestTaskListSerializer(tasks, many=True)
        return Response({'code': 200, 'data': serializer.data})
```

### 5.3 ViewSet 规范

1. **命名**: 使用 `{Model}ViewSet` 格式
2. **queryset**: 使用 `select_related` 和 `prefetch_related` 优化查询
3. **序列化器**: 根据动作动态选择
4. **自定义动作**: 使用 `@action` 装饰器
5. **响应格式**: 统一使用 `{code, message, data}` 格式
