# 自定义报表

本文档详细介绍如何在 LIMIS 系统中定制报告模板，包括模板结构、Word模板编辑、变量定义和使用、表格和图表配置，以及签名位置配置。

## 1. 报告模板结构

### 1.1 模板层级结构

LIMIS 系统的报告模板采用层级结构：

```
报告模板 (ReportTemplate)
    ├── 基本信息
    │   ├── 模板名称
    │   ├── 模板编码
    │   ├── 适用检测类型
    │   └── 版本信息
    ├── 页面设置
    │   ├── 纸张大小
    │   ├── 页边距
    │   ├── 页眉页脚
    │   └── 打印方向
    ├── 内容区域
    │   ├── 封面
    │   ├── 正文
    │   ├── 检测结果表
    │   └── 附件
    └── 签章区域
        ├── 编制签名
        ├── 审核签名
        └── 批准签名
```

## 2. Word模板编辑

### 2.1 创建Word模板

**步骤：**

1. 打开 Microsoft Word 或 WPS
2. 创建新文档
3. 设置页面格式：
   - 纸张大小：A4
   - 页边距：上下2.5cm，左右2.5cm
   - 页眉：1.5cm
   - 页脚：1.5cm

4. 设计模板布局
5. 插入变量占位符
6. 保存为 `.docx` 格式

### 2.2 页面设置

**页面设置代码示例（python-docx）：**

```python
from docx import Document
from docx.shared import Cm, Mm
from docx.enum.section import WD_ORIENT

def setup_page(doc: Document):
    """设置页面格式"""
    section = doc.sections[0]
    
    # 纸张大小 A4
    section.page_width = Mm(210)
    section.page_height = Mm(297)
    
    # 页边距
    section.top_margin = Cm(2.5)
    section.bottom_margin = Cm(2.5)
    section.left_margin = Cm(2.5)
    section.right_margin = Cm(2.5)
    
    # 页眉页脚距离
    section.header_distance = Cm(1.5)
    section.footer_distance = Cm(1.5)
    
    # 打印方向（纵向）
    section.orientation = WD_ORIENT.PORTRAIT
```

## 3. 变量定义和使用

### 3.1 变量类型

| 变量类型 | 说明 | 示例 |
|----------|------|------|
| 简单变量 | 单个值 | `{{report_no}}` |
| 对象变量 | 嵌套对象 | `{{client.name}}` |
| 列表变量 | 数组循环 | `{% for item in items %}` |
| 条件变量 | 条件判断 | `{% if condition %}` |
| 格式化变量 | 格式化输出 | `{{date|date:'Y-m-d'}}` |
| 计算变量 | 表达式计算 | `{{value|round:2}}` |

### 3.5 变量渲染引擎

```python
# apps/reports/template_engine.py

import re
from typing import Any, Dict
from decimal import Decimal
from datetime import datetime, date

class TemplateEngine:
    """模板渲染引擎"""
    
    # 变量模式
    VARIABLE_PATTERN = r'\{\{\s*(\w+(?:\.\w+)*)\s*\}\}'
    # 循环模式
    FOR_PATTERN = r'\{%\s*for\s+(\w+)\s+in\s+(\w+(?:\.\w+)*)\s*%\}(.*?)\{%\s*endfor\s*%\}'
    # 条件模式
    IF_PATTERN = r'\{%\s*if\s+(.+?)\s*%\}(.*?)\{%\s*endif\s*%\}'
    
    def render(self, template: str, context: Dict[str, Any]) -> str:
        """
        渲染模板
        
        :param template: 模板字符串
        :param context: 变量上下文
        :return: 渲染后的字符串
        """
        result = template
        
        # 处理循环
        result = self._render_for(result, context)
        
        # 处理条件
        result = self._render_if(result, context)
        
        # 处理变量
        result = self._render_variables(result, context)
        
        return result
    
    def _render_variables(self, template: str, context: Dict) -> str:
        """渲染变量"""
        def replace_var(match):
            var_path = match.group(1)
            value = self._get_value(var_path, context)
            return self._format_value(value)
        
        return re.sub(self.VARIABLE_PATTERN, replace_var, template)
    
    def _render_for(self, template: str, context: Dict) -> str:
        """渲染循环"""
        def replace_for(match):
            item_name = match.group(1)
            list_path = match.group(2)
            item_template = match.group(3)
            
            items = self._get_value(list_path, context)
            if not items:
                return ''
            
            result = []
            for item in items:
                item_context = {**context, item_name: item}
                result.append(self._render_variables(item_template, item_context))
            
            return ''.join(result)
        
        return re.sub(self.FOR_PATTERN, replace_for, template, flags=re.DOTALL)
    
    def _render_if(self, template: str, context: Dict) -> str:
        """渲染条件"""
        def replace_if(match):
            condition = match.group(1)
            content = match.group(2)
            
            if self._evaluate_condition(condition, context):
                return content
            return ''
        
        return re.sub(self.IF_PATTERN, replace_if, template, flags=re.DOTALL)
    
    def _get_value(self, path: str, context: Dict) -> Any:
        """获取变量值"""
        keys = path.split('.')
        value = context
        
        for key in keys:
            if isinstance(value, dict):
                value = value.get(key)
            elif hasattr(value, key):
                value = getattr(value, key)
            else:
                return None
            
            if value is None:
                return None
        
        return value
    
    def _format_value(self, value: Any) -> str:
        """格式化值"""
        if value is None:
            return ''
        if isinstance(value, Decimal):
            return str(value)
        if isinstance(value, (datetime, date)):
            return value.strftime('%Y年%m月%d日')
        if isinstance(value, bool):
            return '是' if value else '否'
        return str(value)
    
    def _evaluate_condition(self, condition: str, context: Dict) -> bool:
        """评估条件表达式"""
        # 简单条件判断
        value = self._get_value(condition, context)
        return bool(value)
```

```python
# apps/reports/table_generator.py

from docx import Document
from docx.shared import Pt, Cm, Inches
from docx.enum.text import WD_ALIGN_PARAGRAPH
from docx.enum.table import WD_TABLE_ALIGNMENT
from docx.oxml.ns import qn
from docx.oxml import OxmlElement

class TableGenerator:
    """表格生成器"""
    
    def create_table(self, doc: Document, config: dict, data: list) -> None:
        """
        创建表格
        
        :param doc: Word文档对象
        :param config: 表格配置
        :param data: 表格数据
        """
        columns = config['columns']
        style = config.get('style', {})
        
        # 创建表格
        rows = len(data) + 1  # 数据行 + 表头
        table = doc.add_table(rows=rows, cols=len(columns))
        table.style = 'Table Grid'
        table.alignment = WD_TABLE_ALIGNMENT.CENTER
        
        # 设置表头
        self._set_header(table, columns, style)
        
        # 填充数据
        self._fill_data(table, columns, data, style)
        
        # 设置列宽
        self._set_column_width(table, columns)
    
    def _set_header(self, table, columns: list, style: dict) -> None:
        """设置表头"""
        header_row = table.rows[0]
        
        for idx, col in enumerate(columns):
            cell = header_row.cells[idx]
            cell.text = col['header']
            
            # 设置样式
            para = cell.paragraphs[0]
            para.alignment = WD_ALIGN_PARAGRAPH.CENTER
            
            # 设置字体
            run = para.runs[0]
            run.font.bold = True
            run.font.size = Pt(10.5)
            run.font.name = '宋体'
            run._element.rPr.rFonts.set(qn('w:eastAsia'), '宋体')
            
            # 设置背景色
            if style.get('header_bg'):
                self._set_cell_shading(cell, style['header_bg'])
    
    def _fill_data(self, table, columns: list, data: list, style: dict) -> None:
        """填充数据"""
        for row_idx, row_data in enumerate(data):
            row = table.rows[row_idx + 1]
            
            for col_idx, col in enumerate(columns):
                cell = row.cells[col_idx]
                
                # 获取字段值
                field = col['field']
                if col.get('type') == 'auto_increment':
                    value = str(row_idx + 1)
                else:
                    value = getattr(row_data, field, '')
                    if value is None:
                        value = ''
                
                # 格式化值
                value = self._format_value(value, col)
                
                cell.text = str(value)
                
                # 设置对齐
                align_map = {
                    'left': WD_ALIGN_PARAGRAPH.LEFT,
                    'center': WD_ALIGN_PARAGRAPH.CENTER,
                    'right': WD_ALIGN_PARAGRAPH.RIGHT,
                }
                cell.paragraphs[0].alignment = align_map.get(col.get('align', 'left'))
                
                # 设置字体
                run = cell.paragraphs[0].runs[0]
                run.font.size = Pt(style.get('font_size', 10.5))
                run.font.name = '宋体'
                run._element.rPr.rFonts.set(qn('w:eastAsia'), '宋体')
    
    def _format_value(self, value: Any, col: dict) -> str:
        """格式化值"""
        if value is None:
            return ''
        
        format_type = col.get('format', '')
        
        if format_type.startswith('decimal:'):
            precision = int(format_type.split(':')[1])
            if isinstance(value, (int, float, Decimal)):
                return f'{float(value):.{precision}f}'
        
        if col.get('type') == 'judgment':
            return col.get('success_value', '合格') if value else col.get('fail_value', '不合格')
        
        return str(value)
    
    def _set_column_width(self, table, columns: list) -> None:
        """设置列宽"""
        for idx, col in enumerate(columns):
            width = col.get('width', '100px')
            if width.endswith('px'):
                width = int(width[:-2])
            elif width.endswith('cm'):
                width = Cm(float(width[:-2]))
            else:
                width = Cm(2)
            
            for row in table.rows:
                row.cells[idx].width = Cm(width / 37.8)  # px转cm
    
    def _set_cell_shading(self, cell, color: str) -> None:
        """设置单元格背景色"""
        shading = OxmlElement('w:shd')
        shading.set(qn('w:fill'), color.replace('#', ''))
        cell._tc.get_or_add_tcPr().append(shading)
```

```python
# apps/reports/signature_handler.py

from docx import Document
from docx.shared import Pt, Cm, Inches
from docx.enum.text import WD_ALIGN_PARAGRAPH
from docx.oxml.ns import qn
import os

class SignatureHandler:
    """签名处理器"""
    
    def add_signatures(self, doc: Document, config: dict, context: dict) -> None:
        """
        添加签名区域
        
        :param doc: Word文档对象
        :param config: 签名配置
        :param context: 数据上下文
        """
        positions = config.get('positions', [])
        layout = config.get('layout', 'horizontal')
        
        if layout == 'horizontal':
            self._add_horizontal_signatures(doc, positions, context, config.get('style', {}))
        else:
            self._add_vertical_signatures(doc, positions, context, config.get('style', {}))
    
    def _add_horizontal_signatures(self, doc, positions: list, context: dict, style: dict) -> None:
        """添加水平排列的签名"""
        # 创建签名表格
        table = doc.add_table(rows=3, cols=len(positions))
        table.alignment = WD_TABLE_ALIGNMENT.CENTER
        
        # 移除表格边框
        self._remove_table_border(table)
        
        for idx, pos in enumerate(positions):
            # 第一行：标题
            title_cell = table.rows[0].cells[idx]
            title_cell.text = pos['title']
            title_cell.paragraphs[0].alignment = WD_ALIGN_PARAGRAPH.CENTER
            self._set_font(title_cell, style.get('title_font', {}))
            
            # 第二行：签名图片
            sign_cell = table.rows[1].cells[idx]
            self._add_signature_image(sign_cell, pos, context)
            
            # 第三行：日期
            date_cell = table.rows[2].cells[idx]
            date_value = self._get_value(pos['fields']['date']['source'], context)
            date_cell.text = self._format_date(date_value, pos['fields']['date'].get('format'))
            date_cell.paragraphs[0].alignment = WD_ALIGN_PARAGRAPH.CENTER
            self._set_font(date_cell, style.get('content_font', {}))
    
    def _add_signature_image(self, cell, pos: dict, context: dict) -> None:
        """添加签名图片"""
        signature_field = pos['fields'].get('signature', {})
        source = signature_field.get('source', '')
        
        signature_path = self._get_value(source, context)
        
        if signature_path and os.path.exists(signature_path):
            para = cell.paragraphs[0]
            para.alignment = WD_ALIGN_PARAGRAPH.CENTER
            run = para.add_run()
            run.add_picture(
                signature_path,
                width=Cm(signature_field.get('width', 2)),
                height=Cm(signature_field.get('height', 1))
            )
        else:
            # 如果没有签名图片，显示姓名
            name_field = pos['fields'].get('name', {})
            name = self._get_value(name_field.get('source', ''), context)
            cell.text = name or ''
            cell.paragraphs[0].alignment = WD_ALIGN_PARAGRAPH.CENTER
    
    def _remove_table_border(self, table) -> None:
        """移除表格边框"""
        tbl = table._tbl
        tblPr = tbl.tblPr if tbl.tblPr is not None else OxmlElement('w:tblPr')
        tblBorders = OxmlElement('w:tblBorders')
        
        for border_name in ['top', 'left', 'bottom', 'right', 'insideH', 'insideV']:
            border = OxmlElement(f'w:{border_name}')
            border.set(qn('w:val'), 'nil')
            tblBorders.append(border)
        
        tblPr.append(tblBorders)
    
    def _set_font(self, cell, font_config: dict) -> None:
        """设置字体"""
        para = cell.paragraphs[0]
        for run in para.runs:
            run.font.name = font_config.get('name', '宋体')
            run._element.rPr.rFonts.set(qn('w:eastAsia'), font_config.get('name', '宋体'))
            run.font.size = Pt(font_config.get('size', 10.5))
            run.font.bold = font_config.get('bold', False)
    
    def _get_value(self, path: str, context: dict) -> Any:
        """获取变量值"""
        if not path:
            return None
        keys = path.split('.')
        value = context
        for key in keys:
            if isinstance(value, dict):
                value = value.get(key)
            elif hasattr(value, key):
                value = getattr(value, key)
            else:
                return None
        return value
    
    def _format_date(self, value, format_str: str = None) -> str:
        """格式化日期"""
        if not value:
            return ''
        if format_str:
            format_str = format_str.replace('YYYY', '%Y').replace('MM', '%m').replace('DD', '%d')
            return value.strftime(format_str)
        return value.strftime('%Y年%m月%d日')
```

### 5.3 电子签章

```python
# apps/reports/digital_signature.py

import hashlib
from datetime import datetime
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.primitives.asymmetric import padding
from cryptography.hazmat.backends import default_backend
from cryptography.hazmat.primitives.serialization import load_pem_private_key

class DigitalSignature:
    """电子签章"""
    
    def __init__(self, private_key_path: str, certificate_path: str):
        self.private_key = self._load_private_key(private_key_path)
        self.certificate = self._load_certificate(certificate_path)
    
    def sign_report(self, report_data: bytes) -> dict:
        """
        对报告进行数字签名
        
        :param report_data: 报告数据
        :return: 签名信息
        """
        # 计算哈希
        digest = hashlib.sha256(report_data).digest()
        
        # 使用私钥签名
        signature = self.private_key.sign(
            digest,
            padding.PKCS1v15(),
            hashes.SHA256()
        )
        
        return {
            'signature': signature.hex(),
            'algorithm': 'SHA256withRSA',
            'timestamp': datetime.now().isoformat(),
            'certificate_sn': self.certificate.serial_number,
        }
    
    def verify_signature(self, report_data: bytes, signature_info: dict) -> bool:
        """
        验证签名
        
        :param report_data: 报告数据
        :param signature_info: 签名信息
        :return: 验证结果
        """
        try:
            digest = hashlib.sha256(report_data).digest()
            signature = bytes.fromhex(signature_info['signature'])
            
            self.certificate.public_key().verify(
                signature,
                digest,
                padding.PKCS1v15(),
                hashes.SHA256()
            )
            return True
        except Exception:
            return False
    
    def _load_private_key(self, path: str):
        """加载私钥"""
        with open(path, 'rb') as f:
            return load_pem_private_key(f.read(), password=None, backend=default_backend())
    
    def _load_certificate(self, path: str):
        """加载证书"""
        from cryptography import x509
        with open(path, 'rb') as f:
            return x509.load_pem_x509_certificate(f.read(), default_backend())
```

## 6. 完整报告生成流程

### 6.1 报告生成服务

```python
# apps/reports/generator.py

import io
from datetime import datetime
from typing import Dict, Any

from django.conf import settings
from django.core.files.base import ContentFile
from docx import Document

from .template_engine import TemplateEngine
from .table_generator import TableGenerator
from .chart_generator import ChartGenerator
from .signature_handler import SignatureHandler
from .models import Report, ReportTemplate

class ReportGenerator:
    """报告生成器"""
    
    def __init__(self):
        self.template_engine = TemplateEngine()
        self.table_generator = TableGenerator()
        self.chart_generator = ChartGenerator()
        self.signature_handler = SignatureHandler()
    
    def generate_report(self, report_id: int) -> bytes:
        """
        生成报告
        
        :param report_id: 报告ID
        :return: Word文档字节流
        """
        # 获取报告数据
        report = self._get_report_data(report_id)
        template = report.template
        
        # 加载模板文件
        doc = Document(template.template_file.path)
        
        # 构建上下文
        context = self._build_context(report)
        
        # 渲染变量
        self._render_document(doc, template, context)
        
        # 生成表格
        self._generate_tables(doc, template, context)
        
        # 生成图表
        self._generate_charts(doc, template, context)
        
        # 添加签名
        self._add_signatures(doc, template, context)
        
        # 保存到字节流
        buffer = io.BytesIO()
        doc.save(buffer)
        buffer.seek(0)
        
        return buffer.getvalue()
    
    def _get_report_data(self, report_id: int) -> Report:
        """获取报告数据"""
        return Report.objects.select_related(
            'commission', 'commission__project',
            'compiler', 'auditor', 'approver',
            'template'
        ).prefetch_related(
            'commission__samples',
            'test_results__parameter'
        ).get(pk=report_id)
    
    def _build_context(self, report: Report) -> Dict[str, Any]:
        """构建模板上下文"""
        commission = report.commission
        
        return {
            'report': report,
            'commission': commission,
            'project': commission.project,
            'samples': commission.samples.all(),
            'test_results': report.test_results.all(),
            'compiler': report.compiler,
            'auditor': report.auditor,
            'approver': report.approver,
            'compile_date': report.compile_date,
            'audit_date': report.audit_date,
            'approve_date': report.approve_date,
        }
    
    def _render_document(self, doc: Document, template: ReportTemplate, context: Dict) -> None:
        """渲染文档变量"""
        for paragraph in doc.paragraphs:
            paragraph.text = self.template_engine.render(paragraph.text, context)
        
        for table in doc.tables:
            for row in table.rows:
                for cell in row.cells:
                    for paragraph in cell.paragraphs:
                        paragraph.text = self.template_engine.render(paragraph.text, context)
    
    def _generate_tables(self, doc: Document, template: ReportTemplate, context: Dict) -> None:
        """生成表格"""
        tables_config = template.schema.get('tables', {})
        
        for table_name, table_config in tables_config.items():
            data_source = table_config.get('source')
            data = context.get(data_source, [])
            
            # 找到表格占位符并替换
            for paragraph in doc.paragraphs:
                if f'{{{{{table_name}}}}}' in paragraph.text:
                    # 清除占位符
                    paragraph.clear()
                    # 在此位置插入表格
                    self.table_generator.create_table(doc, table_config, data)
    
    def _generate_charts(self, doc: Document, template: ReportTemplate, context: Dict) -> None:
        """生成图表"""
        charts_config = template.schema.get('charts', {})
        
        for chart_name, chart_config in charts_config.items():
            data_source = chart_config.get('source', 'test_results')
            data = context.get(data_source, [])
            
            # 生成图表
            chart_buffer = self.chart_generator.create_chart(chart_config, data)
            
            # 找到图表占位符并替换
            for paragraph in doc.paragraphs:
                if f'{{{{{chart_name}}}}}' in paragraph.text:
                    paragraph.clear()
                    self.chart_generator.insert_chart_to_doc(doc, chart_buffer)
    
    def _add_signatures(self, doc: Document, template: ReportTemplate, context: Dict) -> None:
        """添加签名"""
        signatures_config = template.schema.get('signatures', {})
        
        if signatures_config:
            self.signature_handler.add_signatures(doc, signatures_config, context)
```

## 7. 最佳实践

### 7.1 模板设计建议

1. **保持简洁**：模板设计应简洁明了，避免过于复杂的格式
2. **变量命名**：使用有意义的变量名，便于维护

4. **测试验证**：新模板上线前进行充分测试

1. **缓存模板**：对常用模板进行缓存
2. **异步生成**：大批量报告使用异步任务生成
3. **资源优化**：压缩图片，减少文件大小

### 7.3 安全考虑

1. **权限控制**：限制模板编辑权限
2. **数据脱敏**：敏感数据在报告中脱敏显示
3. **数字签名**：重要报告使用数字签名保护
