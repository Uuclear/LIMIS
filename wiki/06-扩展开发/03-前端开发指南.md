## 2. 组件开发规范

### 2.1 组件命名规范

| 类型 | 命名规范 | 示例 |
|------|----------|------|
| 页面组件 | PascalCase，功能描述 | `TaskList.vue`、`TaskDetail.vue` |
| 公共组件 | PascalCase，功能描述 | `SignatureUpload.vue` |
| 布局组件 | PascalCase，Layout前缀 | `MainLayout.vue`、`Header.vue` |
| 业务组件 | PascalCase，模块前缀 | `TestingTaskCard.vue` |

### 2.2 组件结构规范

使用 `<script setup>` 语法：

```vue
<!-- 组件模板 -->
<template>
  <div class="component-name">
    <!-- 组件内容 -->
  </div>
</template>

<!-- 组件脚本 -->
<script setup lang="ts">
import { ref, computed, onMounted } from 'vue'
import { useRoute, useRouter } from 'vue-router'
import { ElMessage } from 'element-plus'

// Props 定义
interface Props {
  title: string
  data?: any
}

const props = withDefaults(defineProps<Props>(), {
  title: '',
  data: null,
})

// Emits 定义
interface Emits {
  (e: 'update', value: any): void
  (e: 'submit'): void
}

const emit = defineEmits<Emits>()

// 响应式状态
const loading = ref(false)
const formData = ref({})

// 计算属性
const computedValue = computed(() => {
  return props.data ? props.data.value : 0
})

// 方法
function handleSubmit() {
  emit('submit')
}

// 生命周期
onMounted(() => {
  // 初始化逻辑
})
</script>

<!-- 组件样式 -->
<style scoped lang="scss">
.component-name {
  // 样式定义
}
</style>
```

```vue
<!-- views/testing/tasks/TaskList.vue -->
<script setup lang="ts">
import { ref, computed, onMounted } from 'vue'
import { useRouter } from 'vue-router'
import { ElMessage, ElMessageBox } from 'element-plus'
import { Plus, Search, Refresh } from '@element-plus/icons-vue'
import {
  getTestTaskList,
  assignTask,
  startTask,
  completeTask,
  deleteTask,
} from '@/api/testing'
import type { TestTask } from '@/types/testing'

const router = useRouter()

// 状态
const loading = ref(false)
const tableData = ref<TestTask[]>([])
const total = ref(0)
const currentPage = ref(1)
const pageSize = ref(20)

// 搜索表单
const searchForm = ref({
  task_no: '',
  status: '',
  tester_id: null,
  date_range: [] as string[],
})

// 状态选项
const statusOptions = [
  { label: '待分配', value: 'pending' },
  { label: '已分配', value: 'assigned' },
  { label: '检测中', value: 'testing' },
  { label: '已完成', value: 'completed' },
  { label: '已复核', value: 'reviewed' },
]

// 获取列表数据
async function fetchData() {
  loading.value = true
  try {
    const res = await getTestTaskList({
      page: currentPage.value,
      page_size: pageSize.value,
      ...searchForm.value,
    })
    tableData.value = res.data.items
    total.value = res.data.pagination.total
  } catch (error) {
    ElMessage.error('获取数据失败')
  } finally {
    loading.value = false
  }
}

// 搜索
function handleSearch() {
  currentPage.value = 1
  fetchData()
}

// 重置
function handleReset() {
  searchForm.value = {
    task_no: '',
    status: '',
    tester_id: null,
    date_range: [],
  }
  handleSearch()
}

// 分页
function handlePageChange(page: number) {
  currentPage.value = page
  fetchData()
}

function handleSizeChange(size: number) {
  pageSize.value = size
  currentPage.value = 1
  fetchData()
}

// 新建
function handleCreate() {
  router.push('/testing/tasks/new')
}

// 查看详情
function handleView(row: TestTask) {
  router.push(`/testing/tasks/${row.id}`)
}

// 分配任务
async function handleAssign(row: TestTask) {
  try {
    const { value } = await ElMessageBox.prompt('请输入检测员ID', '分配任务', {
      confirmButtonText: '确定',
      cancelButtonText: '取消',
      inputPattern: /^[0-9]+$/,
      inputErrorMessage: '请输入有效的ID',
    })
    await assignTask(row.id, { tester: Number(value) })
    ElMessage.success('分配成功')
    fetchData()
  } catch {
    // 用户取消
  }
}

// 开始检测
async function handleStart(row: TestTask) {
  try {
    await ElMessageBox.confirm('确认开始检测？', '提示')
    await startTask(row.id)
    ElMessage.success('已开始检测')
    fetchData()
  } catch {
    // 用户取消
  }
}

// 完成检测
async function handleComplete(row: TestTask) {
  try {
    await ElMessageBox.confirm('确认完成检测？', '提示')
    await completeTask(row.id)
    ElMessage.success('检测完成')
    fetchData()
  } catch {
    // 用户取消
  }
}

// 删除
async function handleDelete(row: TestTask) {
  try {
    await ElMessageBox.confirm('确认删除该任务？', '警告', {
      type: 'warning',
    })
    await deleteTask(row.id)
    ElMessage.success('删除成功')
    fetchData()
  } catch {
    // 用户取消
  }
}

// 获取状态标签类型
function getStatusType(status: string) {
  const typeMap: Record<string, string> = {
    pending: 'info',
    assigned: 'warning',
    testing: 'primary',
    completed: 'success',
    reviewed: 'success',
  }
  return typeMap[status] || 'info'
}

// 初始化
onMounted(() => {
  fetchData()
})
</script>

<template>
  <div class="page-container">
    <!-- 搜索栏 -->
    <el-card shadow="never" class="search-card">
      <el-form :model="searchForm" inline>
        <el-form-item label="任务编号">
          <el-input
            v-model="searchForm.task_no"
            placeholder="请输入任务编号"
            clearable
            @keyup.enter="handleSearch"
          />
        </el-form-item>
        <el-form-item label="状态">
          <el-select
            v-model="searchForm.status"
            placeholder="请选择状态"
            clearable
          >
            <el-option
              v-for="item in statusOptions"
              :key="item.value"
              :label="item.label"
              :value="item.value"
            />
          </el-select>
        </el-form-item>
        <el-form-item label="检测日期">
          <el-date-picker
            v-model="searchForm.date_range"
            type="daterange"
            range-separator="至"
            start-placeholder="开始日期"
            end-placeholder="结束日期"
            value-format="YYYY-MM-DD"
          />
        </el-form-item>
        <el-form-item>
          <el-button type="primary" :icon="Search" @click="handleSearch">
            搜索
          </el-button>
          <el-button :icon="Refresh" @click="handleReset">
            重置
          </el-button>

    <!-- 操作栏 -->

      </el-button>
    </div>

    <!-- 数据表格 -->

      <el-table

      >

        <el-table-column prop="sample_name" label="样品名称" min-width="150" />
        <el-table-column prop="method_name" label="检测方法" min-width="200" />
        <el-table-column prop="tester_name" label="检测员" width="100" />
        <el-table-column prop="planned_date" label="计划日期" width="120" />

        </el-table-column>
        <el-table-column label="超期" width="80" align="center">

              查看

              link
              type="primary"
              @click="handleAssign(row)"
            >
              分配
            </el-button>
            <el-button
              v-if="row.status === 'assigned'"
              link
              type="primary"
              @click="handleStart(row)"

              开始

            <el-button
              v-if="row.status === 'testing'"
              link

              完成
```

```vue
<!-- views/testing/records/RecordForm.vue -->
<script setup lang="ts">
import { ref, computed, onMounted } from 'vue'
import { useRoute, useRouter } from 'vue-router'
import { ArrowLeft, Plus, Delete } from '@element-plus/icons-vue'
import { ElMessage, ElMessageBox } from 'element-plus'
import {
  getOriginalRecord,
  createOriginalRecord,
  updateOriginalRecord,
  submitRecord,
} from '@/api/testing'
import type { OriginalRecord } from '@/types/testing'

const route = useRoute()
const router = useRouter()

// 状态
const loading = ref(false)
const saving = ref(false)

// 计算属性
const recordId = computed(() => {
  const id = route.params.id as string
  return id === 'new' ? null : Number(id)
})

const isEdit = computed(() => recordId.value != null)
const isReadonly = computed(() => {
  if (!record.value) return false
  return record.value.status !== 'draft'
})

// 表单数据
const form = ref({
  task_id: null as number | null,
  template_id: null as number | null,
  environment_temp: null as number | null,
  environment_humidity: null as number | null,
  data: {
    rows: [] as DataRow[],
  },
})

interface DataRow {
  parameter_name: string
  unit: string
  value_1: string
  value_2: string
  value_3: string
  average: string
  remark: string
}

const record = ref<OriginalRecord | null>(null)

// 获取记录详情
async function fetchRecord() {
  if (!recordId.value) return
  loading.value = true
  try {
    const res = await getOriginalRecord(recordId.value)
    record.value = res
    form.value = {
      task_id: res.task_id,
      template_id: res.template_id,
      environment_temp: res.environment_temp,
      environment_humidity: res.environment_humidity,
      data: res.data ?? { rows: [] },
    }
  } finally {
    loading.value = false
  }
}

// 添加数据行
function addRow() {
  form.value.data.rows.push({
    parameter_name: '',
    unit: '',
    value_1: '',
    value_2: '',
    value_3: '',
    average: '',
    remark: '',
  })
}

// 删除数据行
function removeRow(index: number) {
  form.value.data.rows.splice(index, 1)
}

// 保存
async function handleSave() {
  saving.value = true
  try {
    if (isEdit.value && recordId.value) {
      await updateOriginalRecord(recordId.value, form.value)
      ElMessage.success('保存成功')
    } else {
      const res = await createOriginalRecord(form.value)
      ElMessage.success('创建成功')
      router.replace(`/testing/records/${res.id}`)
    }
  } catch {
    ElMessage.error('保存失败')
  } finally {
    saving.value = false
  }
}

// 提交审核
async function handleSubmit() {
  const id = recordId.value ?? record.value?.id
  if (!id) {
    ElMessage.warning('请先保存记录')
    return
  }
  try {
    await ElMessageBox.confirm('确认提交审核？提交后不可修改。', '提示')
    await submitRecord(id)
    ElMessage.success('提交成功')
    router.push('/testing/records')
  } catch {
    // 用户取消
  }
}

// 返回
function goBack() {
  router.push('/testing/records')
}

// 初始化
onMounted(() => {
  if (isEdit.value) {
    fetchRecord()
  } else {
    addRow()
  }
})
</script>

<template>
  <div class="page-container" v-loading="loading">
    <!-- 页面头部 -->
    <div class="page-header">
      <el-button :icon="ArrowLeft" @click="goBack">返回</el-button>
      <span class="page-title">
        {{ isEdit ? '编辑原始记录' : '新建原始记录' }}
      </span>
      <el-tag v-if="record" :type="record.status === 'draft' ? 'info' : 'success'">
        {{ record.status_display }}
      </el-tag>
    </div>

    <!-- 基本信息 -->
    <el-card shadow="never" style="margin-top: 16px">
      <template #header>基本信息</template>
      <el-form :model="form" label-width="100px" :disabled="isReadonly">
        <el-row :gutter="20">
          <el-col :span="8">
            <el-form-item label="任务ID">

          </el-col>
          <el-col :span="8">
```

## 4. 状态管理 (Pinia)

### 4.1 用户状态

```typescript
// src/stores/user.ts

import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type { UserInfo, Role } from '@/types/user'
import { getToken, setToken, removeToken } from '@/utils/auth'
import * as authApi from '@/api/auth'

export const useUserStore = defineStore('user', () => {
  // 状态
  const userInfo = ref<UserInfo | null>(null)
  const token = ref<string | null>(getToken())
  const permissions = ref<string[]>([])
  const roles = ref<Role[]>([])

  // 计算属性
  const isAuthenticated = computed(() => !!token.value)
  const userName = computed(() => userInfo.value?.name || userInfo.value?.username || '')
  const userRoles = computed(() => roles.value.map(r => r.code))

  // 登录
  async function login(username: string, password: string) {
    const res = await authApi.login(username, password)
    token.value = res.access
    setToken(res.access, res.refresh)
    if (res.user) {
      userInfo.value = res.user
      roles.value = res.user.roles || []
    }
    await getUserInfo()
  }

  // 获取用户信息
  async function getUserInfo() {
    try {
      const res = await authApi.getCurrentUser()
      userInfo.value = res as UserInfo
      permissions.value = (res as any).permissions || []
      roles.value = (res as any).roles || []
    } catch {
      // 静默处理
    }
  }

  // 登出
  async function logout() {
    try {
      await authApi.logout()
    } finally {
      resetState()
    }
  }

  // 刷新 Token
  async function refreshUserToken() {
    const refreshTk = localStorage.getItem('lims_refresh_token')
    if (!refreshTk) {
      resetState()
      return
    }
    try {
      const res = await authApi.refreshToken(refreshTk)
      token.value = res.access
      setToken(res.access, refreshTk)
    } catch {
      resetState()
    }
  }

  // 重置状态
  function resetState() {
    token.value = null
    userInfo.value = null
    permissions.value = []
    roles.value = []
    removeToken()
  }

  // 检查权限
  function hasPermission(permission: string): boolean {
    if (permissions.value.includes('*')) return true
    return permissions.value.includes(permission)
  }

  // 检查角色
  function hasRole(role: string): boolean {
    return userRoles.value.includes(role)
  }

  return {
    // 状态
    userInfo,
    token,
    permissions,
    roles,
    // 计算属性
    isAuthenticated,
    userName,
    userRoles,
    // 方法
    login,
    logout,
    getUserInfo,
    refreshUserToken,
    resetState,
    hasPermission,
    hasRole,
  }
})
```

### 4.2 通知状态

```typescript
// src/stores/notification.ts

import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { getNotifications, getUnreadCount, markAsRead } from '@/api/notification'

export const useNotificationStore = defineStore('notification', () => {
  const notifications = ref<any[]>([])
  const unreadCount = ref(0)
  const loading = ref(false)

  // 获取通知列表
  async function fetchNotifications(params?: any) {
    loading.value = true
    try {
      const res = await getNotifications(params)
      notifications.value = res.data.items
      unreadCount.value = res.data.unread_count
    } finally {
      loading.value = false
    }
  }

  // 获取未读数量
  async function fetchUnreadCount() {
    const res = await getUnreadCount()
    unreadCount.value = res.data.total
  }

  // 标记已读
  async function markRead(id: number) {
    await markAsRead(id)
    unreadCount.value = Math.max(0, unreadCount.value - 1)
    const notification = notifications.value.find(n => n.id === id)
    if (notification) {
      notification.is_read = true
    }
  }

  // 清空通知
  function clearNotifications() {
    notifications.value = []
    unreadCount.value = 0
  }

  return {
    notifications,
    unreadCount,
    loading,
    fetchNotifications,
    fetchUnreadCount,
    markRead,
    clearNotifications,
  }
})
```

### 5.1 Axios 封装

```typescript
// src/utils/request.ts

import axios from 'axios'
import type { AxiosInstance, AxiosRequestConfig, AxiosResponse } from 'axios'
import { ElMessage } from 'element-plus'
import { getToken, removeToken } from '@/utils/auth'
import router from '@/router'

// 创建 axios 实例
const service: AxiosInstance = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL || '/api/v1',
  timeout: 30000,
  headers: {
    'Content-Type': 'application/json',
  },
})

// 请求拦截器
service.interceptors.request.use(
  (config) => {
    // 添加 Token
    const token = getToken()
    if (token) {
      config.headers.Authorization = `Bearer ${token}`
    }
    return config
  },
  (error) => {
    console.error('请求错误:', error)
    return Promise.reject(error)
  }
)

// 响应拦截器
service.interceptors.response.use(
  (response: AxiosResponse) => {
    const { data } = response

    // 业务错误处理
    if (data.code && data.code !== 200) {
      ElMessage.error(data.message || '请求失败')
      return Promise.reject(new Error(data.message || 'Error'))
    }

    return data
  },
  (error) => {
    const { response } = error

    if (response) {
      switch (response.status) {
        case 401:
          // Token 过期，跳转登录
          removeToken()
          router.push('/login')
          ElMessage.error('登录已过期，请重新登录')
          break
        case 403:
          ElMessage.error('没有权限访问')
          break
        case 404:
          ElMessage.error('请求的资源不存在')
          break
        case 500:
          ElMessage.error('服务器内部错误')
          break
        default:
          ElMessage.error(response.data?.message || '请求失败')
      }
    } else {
      ElMessage.error('网络连接失败')
    }

    return Promise.reject(error)
  }
)

// 通用请求方法
export function request<T = any>(config: AxiosRequestConfig): Promise<T> {
  return service.request<any, T>(config)
}

export default service
```
